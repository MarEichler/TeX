\subsection{Spacing}
\subsubsection{Definitions}
\begin{itemize}
	\item \code{\textbackslash itemsep} vertical space added \textbf{after} each item in the list
	\item \code{\textbackslash topsep} vertical space added \textbf{above and below} the list 
	\item \code{\textbackslash parsep} vertical space added \textbf{after} each paragraph in the list, I think each item is treated like a new paragraph so I usually add both \code{\textbackslash itemsep} and \code{\textbackslash parsep}
	\item \code{\textbackslash parttopsep} verticale space added \textbf{above and below} the list, but only if the lists starts a new paragraph 
\end{itemize}
\subsubsection{Visual Example}
